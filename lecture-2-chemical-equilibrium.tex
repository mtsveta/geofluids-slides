\section{Chemical equilibrium from a thermodynamic and mathematical perspective}

\begin{frame}[<+->]{Chemical equilibrium from a thermodynamic perspective}
\begin{itemize}
\item A system under prescribed temperature $T$ and pressure $P$ is in
chemical equilibrium if its \textbf{Gibbs energy}, $G$, is at a minimum. 
\begin{itemize}
\item this is a result of the \emph{second law of thermodynamic} that says
that a system tends to a state of maximum entropy $S$. 
\item the minimum Gibbs energy condition arises because $G=H-TS$, where
$H$ is enthalpy. 
\end{itemize}
\item Compare this with a system in \textbf{mechanical equilibrium}. For
example, a \emph{ball rolling down a hill}. 
\item In this analog problem, we want to find the \textbf{final position}
of the ball that results in a minimum for its \textbf{potential energy}.
\item Here, we want to find the \textbf{amounts of each species}, in every
phase, that \textbf{minimizes the Gibbs energy }of the chemical system.
\end{itemize}
\end{frame}
%
\begin{frame}[<+->]{Constraints in a chemical equilibrium state}
\begin{itemize}
\item A ball rolling down a hill is \textbf{constrained by its surface topology}. 
\item Similarly, the species in a chemical system seeking a state of chemical
equilibrium is also \textbf{constrained by some conditions}: \textbf{\alert{\textbf{mass conservation}}}.
\end{itemize}
\end{frame}
%

# Transform Python lists A and n into Numpy arrays
A = numpy.array(A)
n = numpy.array(n)

# Multiply matrix A and vector n to calculate the amounts of elements, b
b = A @ n

# Create a list with the names of the elements
elements = ['H', 'O', 'C', 'Z']

# Create a list with the molar masses of H, O, C, Z
molar_masses_elements = [1.0079, 15.9994, 12.0107, 0.0]

# Loop over all elements, their amounts, and their molar masses
for element, amount, molarmass in zip(elements, b, molar_masses_elements):
    print('Element %s has %f moles and %f grams' % (element, amount, amount*molarmass))

print() # Just to skip one line in the output

# Alternative way of looping over all species using an index i
for i in range(len(species)):
    # Calculate the molar mass of current species
    molar_mass = A[:, i] @ molar_masses_elements
    # Calculate the mass of current species
    mass = n[i] * molar_mass
    print('Species {} has {} moles and {} grams.'.format(species[i], n[i], mass))

\begin{frame}{Chemical equilibrium formulated as Gibbs energy minimization problem}

\begin{cbox}{Gibbs energy minimization (GEM) problem}

Given temperature $T$, pressure $P$, and amounts of elements ${b=(b_{1},\ldots,b_{\text{E}})}$,
find the amounts of species ${n=(n_{1},\ldots,n_{\text{N}})}$ that
solve the \textbf{constrained minimization problem}:
\[
\min_{n}G(n;T,P)\qquad\text{subject to}\qquad An=b\quad\text{and}\quad n\geq0.
\]

\end{cbox}where:\hiddenpause
\begin{itemize}
\item $G$ is the Gibbs energy of the system, a function of $n=(n_{1},\ldots,n_{\text{N}})$
as well as temperature $T$ and pressure $P$;\hiddenpause
\item $An=b$ are the mass conservation constraints for each element;\hiddenpause
\item $n\geq0$ are the non-negative constraints for the amounts of each
species.
\end{itemize}
\end{frame}
%
\begin{frame}{Gibbs energy function}

The Gibbs energy function, $G$, is defined as: 
\[
G=\sum_{i=1}^{\text{N}}n_{i}\mu_{i},
\]
where $\mu_{i}$ is the \alert{chemical potential} of the $i$th
species. 
\begin{itemize}
\item \textbf{Question:} What is the dimension (in SI units) of $G$ and
$\mu_{i}$? 
\item \visible<2>{\textbf{Answer:} $G$ has units of J, and $\mu_{i}$ units of J/mol.}
\end{itemize}
\end{frame}
%
\begin{frame}{Chemical potential and its interpretation}

\begin{itemize}[<+->]
\item The chemical potential of a substance, $\mu_{i}$, is a measure of
its tendency to transform to other substances.
\item In the following reaction:
\[
A\rightleftharpoons B,
\]

\begin{itemize}
\item $\mu_{A}>\mu_{B}$: substance $A$ will tend to transform to $B$ 
\item $\mu_{A}=\mu_{B}$: substances $A$ and $B$ are in equilibrium, no
transformation 
\item $\mu_{A}<\mu_{B}$: substance $B$ will tend to transform to $A$
\end{itemize}
\item The higher the value of $\mu_{A}$, the higher the tendency for A
to transform to B.
\end{itemize}
\end{frame}
%

\begin{frame}{Chemical potentials and direction of reactions}

\begin{itemize}[<+->]
\item The chemical potentials of substances, $\mu_{i}$, can be used to
indicate which direction a reaction should proceed: 
\[
\mathrm{A+B\rightleftharpoons C+2D}
\]
\item $\mu_{A}+\mu_{B}>\mu_{C}+2\mu_{D}$: substances $A$ and $B$ will
tend to transform to $C$ and $D$ 
\item $\mu_{A}+\mu_{B}=\mu_{C}+2\mu_{D}$: substances $A$, $B$, $C$,
and $D$ are in equilibrium 
\item $\mu_{A}+\mu_{B}<\mu_{C}+2\mu_{D}$: substances $C$ and $D$ will
tend to transform to $A$ and $B$
\item \alert{Warning:} Note the coefficients in the reactions and its use
when checking the direction of the reaction!
\[
\boldsymbol{1}A+\boldsymbol{1}B\rightleftharpoons\boldsymbol{1}C+\boldsymbol{2}D
\]
\end{itemize}
\end{frame}

\begin{frame}[<+->]{Chemical potential $\mu_{i}$ and Gibbs energy}
\begin{itemize}
\item The chemical potential of a species, $\mu_{i}$, is equivalent to:{\Large{}
\[
\mu_{i}\equiv\left[\frac{\partial G}{\partial n_{i}}\right]_{T,P}
\]
}{\Large\par}
\item Thus, the infinitesimal change in Gibbs energy, $\mathrm{d}G$, following
an infinitesimal change in the amount of species $i$, $\mathrm{d}n_{i}$,
is given by:
\[
\mathrm{d}G=\mu_{i}\mathrm{d}n_{i}
\]
at constant temperature $T$ and pressure $P$.
\end{itemize}
\end{frame}
%
\begin{frame}{Chemical potential $\mu_{i}$ and its definition}
\begin{itemize}
\item The chemical potential, $\mu_{i}$, of a species is defined as:{\Large{}
\[
\mu_{i}=\mu_{i}^{\circ}+RT\ln a_{i}
\]
}{\Large\par}
\item $\mu_{i}^{\circ}$ is the \alert{standard chemical potential} of
the species at $(T,P)$;
\item $a_{i}$ is the \alert{activity} of the species at $(T,P,n_{\pi})$,
where $n_{\pi}$ is the vector of amounts of species in the phase
where species $i$ lives;
\begin{itemize}
\item for example, the activity of an aqueous species only depends on the
amounts of the aqueous species, and not on the amounts of gases, minerals,
etc.
\end{itemize}
\item $R$ is the \alert{universal gas constant}, $R=\unit[8.314]{J/(mol\cdot K)}$;
\end{itemize}
\end{frame}
%
\begin{frame}{Standard chemical potential $\mu_{i}^{\circ}$}
\begin{itemize}[<+->]
\item The \textbf{standard chemical potential} $\mu_{i}^{\circ}$ of a
species is its chemical potential at a \alert{\emph{standard reference state}},
in which $a_{i}=1$ is assumed.
\item For example, for a gas, $\mu_{i}^{\circ}$ is the chemical potential
of the gas at a hypothetical ideal state at $P^{\circ}=\unit[1]{bar}$.
\item Thus, for a gas, $\mu_{i}^{\circ}=\mu_{i}^{\circ}(T)$ (i.e., \textbf{no
pressure dependence}).
\item We assume here ${\mu_{i}^{\circ}=\mu_{i}^{\circ}(T,P)}$ in general
(true for aqueous and mineral species).
\item Values for $\mu_{i}^{\circ}$ are calculated using equations of state,
which can be computationally expensive.
\item Their values are then saved in tables that can be used for interpolation.
\end{itemize}
\end{frame}

\begin{frame}{Mass action equation and equilibrium constant of reaction}

\lcol
\begin{itemize}
\item Consider the following reaction:
\[
\mathrm{A+2B\rightleftharpoons C+3D}.
\]
\item At equilibrium, the following \alert{mass action equation} must be
satisfied:
\[
K=\frac{a_{C}a_{D}^{3}}{a_{A}a_{B}^{2}},
\]
where $K=K(T,P)$ is the equilibrium constant of the reaction.
\end{itemize}
\rcol
\begin{itemize}
\item We assume $n_{\mathrm{A}},n_{\mathrm{B}},n_{\mathrm{C}},n_{\mathrm{D}}>0$
(i.e., these species are \emph{stable at equilibrium}).
\item For a general reaction:
\[
\sum_{r}^{\text{reactants}}\nu_{r}A_{r}\rightleftharpoons\sum_{p}^{\text{products}}\nu_{p}A_{p},
\]
the mass action equation is:
\[
K=\dfrac{\prod_{p}a_{p}^{\nu_{p}}}{\prod_{r}a_{r}^{\nu_{r}}}.
\]
\end{itemize}
\ecol
\end{frame}

\begin{frame}{Mass action equation and equilibrium constant of reaction}
\begin{itemize}
\item \textbf{Exercise:} At 25 °C and 1 bar, the reaction: 
\[
\mathsf{H_{2}O\text{(aq)}}\rightleftharpoons\mathsf{H^{+}\text{(aq)}+OH^{-}\text{(aq)}}
\]
has $K\approx10^{-14}$. If we have pure water, with no dissolved
solutes other than H$^{+}$(aq) and OH$^{-}$(aq), then it follows
that:
\[
a_{\mathsf{H^{+}\text{(aq)}}}\approx10^{-7}\qquad a_{\mathsf{OH^{-}\text{(aq)}}}\approx10^{-7}\qquad a_{\mathsf{H_{2}O\text{(aq)}}}\approx1.
\]

\begin{enumerate}
\item[{\small{}(a)}] Show that the mass action equation above is satisfied with the above
conditions.
\item[{\small{}(b)}] What is the pH of water at those conditions, where $\text{pH}\coloneqq-\log a_{\mathsf{H^{+}\text{(aq)}}}$?
\end{enumerate}
\end{itemize}
\end{frame}

\begin{frame}[<+->]{Equilibrium constant of reactions from standard chemical potentials}
\begin{itemize}
\item The \textbf{\alert{equilibrium constant of a reaction}}, $K$, is
defined as:
\[
\ln K=-\frac{1}{RT}\left[\sum_{p}\nu_{p}\mu_{p}^{\circ}-\sum_{r}\nu_{r}\mu_{r}^{\circ}\right],
\]
where $R=\unit[8.314]{\mathsf{J/(mol\cdot K)}}$ is the universal
gas constant; $T$ is temperature (in K). 
\item \textbf{Exercise:} At 100 °C and 50 bar, ${\mu_{\mathsf{H_{2}O\text{(aq)}}}^{\circ}=\unit[-242.992]{\mathsf{kJ/mol}}}$,
${\mu_{\mathsf{H^{+}\text{(aq)}}}^{\circ}=\unit[0.0]{\mathsf{kJ/mol}}}$,
and ${\mu_{\mathsf{OH^{-}\text{(aq)}}}^{\circ}=\unit[-155.559]{\mathsf{kJ/mol}}}$.
Calculate $\ln K$ and $\log K$ for the reaction:
\[
\mathsf{H_{2}O\text{(aq)}}\rightleftharpoons\mathsf{H^{+}\text{(aq)}+OH^{-}\text{(aq)}}.
\]
\hiddenpause
\item \textbf{Answer: $\ln K=-28.18$} and $\log K=-12.18$.
\end{itemize}
\end{frame}
